Введение

Атомарность

Порядок

Чтение может быть переупорядочено с ранее произведенной записью для различных ячеек памяти

Процессор 0 Процессор 1 
mov [_x], 1
mov r1, [_y]
 
mov [_y], 1
mov r2, [_x]
Начальные значения: x = y = 0
r1 = 0 и r2 = 0 разрешено

Однако переупорядочивание чтения с предидущей записью для одной ячейки памяти - запрещено.

Процессор 0 
mov [_x], 1
mov r1, [_x]
Начальные значения: x = 0
r1 = 0 запрещено

Предтвращение переупорядочивания компилятором 

asm volatile("" ::: "memory");

Предотвращение переупорядочивания CPU 

asm volatile("mfence" ::: "memory");

Пример 

/*
gcc -std=gnu11 -O2 -lpthread reordering.c -o reordering 
*/
#include <stdio.h>
#include <pthread.h>

pthread_barrier_t barrier;

volatile int X, Y;
int r1, r2;

void* threadA_function(void* arg)
{
    pthread_barrier_wait(&barrier);
    X = 1;
    asm volatile("" ::: "memory"); // prevent compiler reordering
    r1 = Y;
    return NULL;
}

void* threadB_function(void* arg)
{
    pthread_barrier_wait(&barrier);
    Y = 1;
    asm volatile("" ::: "memory"); // prevent compiler reordering
    r2 = X;
    return NULL;
}

int main()
{
    pthread_t thrA, thrB;

    for (int count=0; ; count++) {
        
        // initial state
        X = Y = 0;
        pthread_barrier_init(&barrier, NULL, 2);

        pthread_create(&thrA, NULL, threadA_function, NULL);
        pthread_create(&thrB, NULL, threadB_function, NULL);

        pthread_join(thrA, NULL);
        pthread_join(thrB, NULL);

        if( (r1 == 0) && (r2 == 0) ){
            printf("%7u> r1=%d, r2=%d (X=%d, Y=%d)\n", count, r1, r2, X, Y);
        } 
   }
   return 0;
}

Видимость

Процессор 0 Процессор 1 
mov [_x], 1
mov r1, [_x]
mov r2, [_y]
 
mov [_y], 1
mov r3, [_y]
mov r4, [_x]
Начальные значения: x = y = 0
r2 = 0 и r4 = 0 разрешено

Пример 

/*
gcc -std=gnu11 -O2 -lpthread visibility.c -o visibility 
*/
#include <stdio.h>
#include <pthread.h>

pthread_barrier_t barrier;

volatile int X, Y;
int r1, r2, r3, r4;

void* threadA_function(void* arg)
{
    pthread_barrier_wait(&barrier);
    X = 1;
    r1 = X;
    r2 = Y;
    return NULL;
}

void* threadB_function(void* arg)
{
    pthread_barrier_wait(&barrier);
    Y = 1;
    r3 = Y;
    r4 = X;
    return NULL;
}

int main()
{
    pthread_t thrA, thrB;

    for (int count=0; ; count++) {
        
        // initial state
        X = Y = 0;
        pthread_barrier_init(&barrier, NULL, 2);

        pthread_create(&thrA, NULL, threadA_function, NULL);
        pthread_create(&thrB, NULL, threadB_function, NULL);

        pthread_join(thrA, NULL);
        pthread_join(thrB, NULL);

        if((r2 == 0) && (r4 == 0)){
            printf("%7u> r2=%d, r4=%d (X=%d, Y=%d)\n", count, r2, r4, X, Y );
        }
   }
   return 0;
}
