Предтвращение переупорядочивания компилятором 

asm volatile("" ::: "memory");
#include<stdio.h>
#include<stdlib.h>
#include<pthread.h>
#include<semaphore.h>

int a, b, X, Y;
sem_t sem_start1, sem_start2, sem_end;

void *thread1_function(){
    printf("Hello i am thread 1\n");
    srand(time(NULL));
    for(;;){
        sem_wait(&sem_start1);
        while(rand() % 8 != 0){}
        X = 1;
        asm volatile("" ::: "memory"); 
        b = Y;
        sem_post(&sem_end);
    }
    return NULL;
}

void *thread2_function(){
    printf("Hello i am thread 2\n");
    srand(time(NULL));
    for(;;){
        sem_wait(&sem_start2);
        while(rand() % 8 !=0 ){}
        Y = 1;
        asm volatile("" ::: "memory"); 
        a = X;
        sem_post(&sem_end);
    }
    return NULL;
}


int main(){
    pthread_t thread1, thread2;
    int res1, res2;
    int it;

    res1 = pthread_create(&thread1, NULL, thread1_function, NULL);
    res2 = pthread_create(&thread2, NULL, thread2_function, NULL);
    if(res1 || res2){
        printf(
            "Could not create threads res1 = %d, res2 = %d\n",
            res1, res2);
        return 1;
    }
    sem_init(&sem_start1, 0, 0);
    sem_init(&sem_start2, 0, 0);
    sem_init(&sem_end, 0, 0);

    for(it=0;; it++){
        X = 0;
        Y = 0;
        sem_post(&sem_start1);
        sem_post(&sem_start2);
        sem_wait(&sem_end);
        sem_wait(&sem_end);
        if(a == 0 && b == 0)
        {
            printf("reorders detected after %d iterations\n", it);
        }
        if(a == 0){
            printf("a == 0\n");
        }
        if(b == 0){
            printf("b == 0\n");
        }
    }
    pthread_join(thread1, NULL);
    pthread_join(thread2, NULL);
    return 0;
}
