do-нотация

Программная запятая через parse_transform

do([Monad ||
    A <- foo(),
    B <- bar(A, dog),
    B
]).

Monad:'>>='(
    foo(),
    fun (A) -> Monad:'>>='(
        bar(A, dog),
        fun (B) -> B end)
    end).

do-нотация (Haskell)

    
    do
        a <- foo
        b <- bar a dog
    

    
    foo >>= \a ->
        bar a dog >>= \b -> b

State монада

Пример реализации вычисления факториала

fact_rec(0) ->
    do([erlmonads_state ||
        Acc <- get(),
        return(Acc)
    ]);
fact_rec(N) ->
    do([erlmonads_state ||
        Acc <- get(),
        put(Acc * N),
        fact_rec(N - 1)
    ]).

fact(N) ->
    erlmonads_state:eval(
        fact_rec(N),
        1
    ).

State монада (Haskell)

Пример реализации вычисления факториала

fact_rec :: Int -> State Int Int
fact_rec 0 = do
    acc <- get
    return acc
fact_rec n = do
    acc <- get
    put (acc * n)
    fact_rec (n - 1)

State монада

Пример реализации вычисления чисел фибоначчи

-import(erlmonads_state, [
    eval/2, modify/1,
    sequence/1, exec/2
]).

fib_m_step() ->
    modify(fun ({X, Y}) -> {Y, X+Y} end).

fib(N) ->
    {A, _} = exec(
        sequence(
            lists:duplicate(N, fib_m_step())
        ),
        {0, 1}
    ),
    A.

State монада (Haskell)

Пример реализации вычисления чисел фибоначчи

fib_m_step :: State (Int, Int) ()
fib_m_step = 
    modify $ \(a,b) -> (b,a+b)

fib :: Int -> Int
fib n = fst ((execState $ 
        sequence (Data.List.replicate n fib_m_step)
    )
    (0, 1))
